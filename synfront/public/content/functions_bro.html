    modifyValue(num);  // num ab 20 ho jayega
    return 0;
}

Recursive Functions - Khud Ko Call Karne Wale! ğŸ”„

Recursive functions woh hote hain jo khud ko hi call karte hain:

int factorial(int n) {
    if (n <= 1) {
        return 1;  // Base case - rukne ki condition
    } else {
        return n * factorial(n - 1);  // Recursive call
    }
}

int main() {
    int result = factorial(5);  // 5! = 5 * 4 * 3 * 2 * 1 = 120
    return 0;
}

Function Pointers - Functions ke Addresses! ğŸ‘‰

Function pointers variables hain jo functions ke addresses store karte hain:

int add(int a, int b) {
    return a + b;
}

int subtract(int a, int b) {
    return a - b;
}

int main() {
    // Function pointer declare karo
    int (*operation)(int, int);
    
    // Function address assign karo
    operation = add;
    cout << operation(5, 3) << endl;  // Output: 8
    
    // Function change karo
    operation = subtract;
    cout << operation(5, 3) << endl;  // Output: 2
    
    return 0;
}
