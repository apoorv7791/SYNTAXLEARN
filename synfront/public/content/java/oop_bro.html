public class Person {
    // Private attributes - bahar se direct access nahi kar sakte
    private String name;
    private int age;
    
    // Constructor
    public Person(String name, int age) {
        this.name = name;
        this.age = age;
    }
    
    // Getter methods - data ko bahar nikalne ke liye
    public String getName() {
        return name;
    }
    
    public int getAge() {
        return age;
    }
    
    // Setter methods - data ko update karne ke liye
    public void setName(String name) {
        this.name = name;
    }
    
    public void setAge(int age) {
        if (age > 0) {  // Validation - negative age nahi set kar sakte!
            this.age = age;
        } else {
            System.out.println("Bhai, age negative nahi ho sakti! ğŸ˜‚");
        }
    }
}

        this.model = model;

        System.out.println("Vehicle start ho raha hai...");

// Subclass (Child class) - Beta
public class Car extends Vehicle {

    public Car(String brand, String model, int numDoors) {

        this.numDoors = numDoors;

    // Override the start method - parent ka method override kiya
    @Override
    public void start() {
        System.out.println(brand + " " + model + " car start ho rahi hai... Vroom vroom!");
    }
    
    // New method specific to Car - car ka apna special method
    public void honk() {
        System.out.println("Horn bajaya... Pom pom! ğŸ“¢");
    }
}

Abstract Classes - Adhure Blueprints! ğŸ—ï¸

// Abstract class - Cannot create objects directly
public abstract class Shape {
    // Abstract method (no implementation) - Subclasses must implement
    public abstract double calculateArea();
    
    // Concrete method (with implementation) - Ready to use
    public void display() {
        System.out.println("Yeh ek shape hai jiska area hai: " + calculateArea());
    }
}

// Concrete subclasses - Complete implementations
public class Circle extends Shape {
    private double radius;
    
    public Circle(double radius) {
        this.radius = radius;
    }
    
    @Override
    public double calculateArea() {
        return Math.PI * radius * radius;
    }
}

public class Rectangle extends Shape {
    private double length;
    private double width;
    
    public Rectangle(double length, double width) {
        this.length = length;
        this.width = width;
    }
    
    @Override
    public double calculateArea() {
        return length * width;
    }
}

Abstract classes ka use kaise karein:

public class Main {
    public static void main(String[] args) {
        // Shape shape = new Shape();  // Error: Cannot instantiate abstract class
        
        Shape circle = new Circle(5);
        Shape rectangle = new Rectangle(4, 6);
        
        circle.display();     // Output: Yeh ek shape hai jiska area hai: 78.53981633974483
        rectangle.display();  // Output: Yeh ek shape hai jiska area hai: 24.0
    }
}
