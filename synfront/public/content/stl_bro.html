Vector - Sabse Popular Container! ğŸŒŸ

#include <vector>
#include <iostream>
using namespace std;

int main() {
    // Vector banao
    vector<int> numbers;  // Int type ka empty vector

    // Elements add karo - peeche se jodna
    numbers.push_back(10);  // [10]
    numbers.push_back(20);  // [10, 20]
    numbers.push_back(30);  // [10, 20, 30]

    // Elements access karo
    cout << "First element: " << numbers[0] << endl;  // 10
    cout << "Second element: " << numbers[1] << endl;  // 20
    cout << "Third element: " << numbers.at(2) << endl;  // 30 (safe access)

    // Vector mein loop karo
    cout << "All elements: ";
    for (int num : numbers) {  // Range-based for loop (C++11)
        cout << num << " ";  // 10 20 30
    }
    cout << endl;

    // Size aur capacity check karo
    cout << "Size: " << numbers.size() << endl;  // 3
    cout << "Capacity: " << numbers.capacity() << endl;  // Kitna space allocated hai

    // Last element hatao
    numbers.pop_back();  // [10, 20]

    // Saare elements clear karo
    numbers.clear();  // []

    return 0;
}

Map - Key-Value Pairs Store Karne Ke Liye! ğŸ—ï¸

#include <map>
#include <string>
#include <iostream>
using namespace std;

int main() {
    // Map banao - phone book jaisa
    map<string, int> ages;  // Name -> Age mapping

    // Elements insert karo
    ages["Alice"] = 30;  // Alice ki age 30
    ages["Bob"] = 25;    // Bob ki age 25
    ages.insert(make_pair("Charlie", 35));  // Charlie ki age 35

    // Elements access karo
    cout << "Alice's age: " << ages["Alice"] << endl;  // 30
    cout << "Bob's age: " << ages.at("Bob") << endl;   // 25 (safe access)

    // Check karo key exist karti hai ya nahi
    if (ages.find("David") == ages.end()) {
        cout << "David not found in the map" << endl;  // David nahi mila
    }

    // Map mein loop karo
    cout << "All ages:" << endl;
    for (const auto& pair : ages) {  // C++11 auto and range-based for
        cout << pair.first << ": " << pair.second << endl;  // Name: Age
    }

    // Size check karo
    cout << "Size: " << ages.size() << endl;  // 3

    // Element remove karo
    ages.erase("Bob");  // Bob ko hatao

    return 0;
}

Algorithms Kaise Use Karte Hain? ğŸ¤”

#include <algorithm>
#include <vector>
#include <iostream>
using namespace std;

int main() {
    vector<int> numbers = {5, 2, 8, 1, 9, 3, 7, 4, 6};  // Unsorted numbers

    // Sort karo - chote se bade
    sort(numbers.begin(), numbers.end());
    cout << "Sorted: ";
    for (int num : numbers) {
        cout << num << " ";  // 1 2 3 4 5 6 7 8 9
    }
    cout << endl;

    // Find karo - element dhundho
    auto it = find(numbers.begin(), numbers.end(), 7);
    if (it != numbers.end()) {
        cout << "Found 7 at position: " << (it - numbers.begin()) << endl;  // Position 6
    }

    // Count karo - kitne 3 hain?
    int count_of_3 = count(numbers.begin(), numbers.end(), 3);
    cout << "Count of 3: " << count_of_3 << endl;  // 1

    // For each - har element ko double karo
    cout << "Doubled values: ";
    for_each(numbers.begin(), numbers.end(), [](int num) {
        cout << num * 2 << " ";  // Har number * 2
    });
    cout << endl;

    // Reverse karo - ulta order
    reverse(numbers.begin(), numbers.end());
    cout << "Reversed: ";
    for (int num : numbers) {
        cout << num << " ";  // 9 8 7 6 5 4 3 2 1
    }
    cout << endl;

    return 0;
}

Iterators Kaise Use Karte Hain? ğŸ¤”

#include <vector>
#include <iostream>
using namespace std;

int main() {
    vector<int> numbers = {10, 20, 30, 40, 50};  // 5 numbers

    // Forward traversal - aage se peeche
    cout << "Forward traversal: ";
    for (vector<int>::iterator it = numbers.begin(); it != numbers.end(); ++it) {
        cout << *it << " ";  // 10 20 30 40 50
    }
    cout << endl;

    // Backward traversal - peeche se aage
    cout << "Backward traversal: ";
    for (vector<int>::reverse_iterator rit = numbers.rbegin(); rit != numbers.rend(); ++rit) {
        cout << *rit << " ";  // 50 40 30 20 10
    }
    cout << endl;

    // Auto use karke (C++11) - typing kam karo!
    cout << "Using auto: ";
    for (auto it = numbers.begin(); it != numbers.end(); ++it) {
        cout << *it << " ";  // 10 20 30 40 50
    }
    cout << endl;

    return 0;
}
