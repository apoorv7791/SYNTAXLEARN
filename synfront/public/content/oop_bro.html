// Class definition - Blueprint banaya
class Car {
private:
    // Data members (attributes) - Car ki properties
    string brand;
    string model;
    int year;

public:
    // Constructor - Car banane ka tarika
    Car(string b, string m, int y) {
        brand = b;
        model = m;
        year = y;
    }

    // Member functions (methods) - Car kya kar sakti hai
    void displayInfo() {
        cout << year << " " << brand << " " << model << endl;
    }

    void accelerate() {
        cout << "The " << brand << " " << model << " is accelerating!" << endl;
    }
};

// Creating objects - Actual cars bana li
Car myCar("Toyota", "Corolla", 2020);  // Meri car
Car yourCar("Honda", "Civic", 2019);   // Teri car

// Using objects - Cars ko use kiya
myCar.displayInfo();  // Output: 2020 Toyota Corolla
yourCar.accelerate();  // Output: The Honda Civic is accelerating!

class BankAccount {
private:
    // Private data members - Ye secret hai bhai!
    double balance;
    string accountNumber;

public:
    // Public methods - Inke through hi data access hoga
    BankAccount(string accNum, double initialBalance) {
        accountNumber = accNum;
        balance = initialBalance;
    }

    void deposit(double amount) {
        if (amount > 0) {
            balance += amount;  // Paise jama karo
        }
    }

    bool withdraw(double amount) {
        if (amount > 0 && balance >= amount) {
            balance -= amount;  // Paise nikalo
            return true;
        }
        return false;  // Balance kam hai bhai!
    }

    double getBalance() {
        return balance;  // Balance check karo
    }

    string getAccountNumber() {
        return accountNumber;
    }
};

Inheritance matlab ek class dusre class ki properties inherit kar sakti hai! Bilkul baap se bete ko properties milne ki tarah!

// Base class - Parent class (Baap)
class Vehicle {
protected:
    string brand;
    int year;

public:
    Vehicle(string b, int y) : brand(b), year(y) {}

    void displayInfo() {
        cout << year << " " << brand << endl;
    }
};

// Derived class - Child class (Beta)
class Car : public Vehicle {
private:
    int doors;

public:
    Car(string b, int y, int d) : Vehicle(b, y), doors(d) {}  // Baap ke properties + apni property

    void displayCarInfo() {
        displayInfo();  // Baap ka method use kiya
        cout << "Doors: " << doors << endl;  // Apna extra info add kiya
    }
};

// Creating and using a derived class object
Car myCar("Toyota", 2020, 4);
myCar.displayCarInfo();
// Output:
// 2020 Toyota
// Doors: 4

Function Overriding - Baap ka Method Override Karna! ğŸ”„

// Base class - Parent
class Shape {
public:
    virtual void draw() {
        cout << "Drawing a shape" << endl;  // General drawing
    }
};

// Derived classes - Children
class Circle : public Shape {
public:
    void draw() override {
        cout << "Drawing a circle" << endl;  // Circle specific drawing
    }
};

class Rectangle : public Shape {
public:
    void draw() override {
        cout << "Drawing a rectangle" << endl;  // Rectangle specific drawing
    }
};

// Polymorphic behavior - Ek tarah se call, alag alag kaam
Shape* shape1 = new Circle();
Shape* shape2 = new Rectangle();

shape1->draw();  // Output: Drawing a circle
shape2->draw();  // Output: Drawing a rectangle

// Clean up - Memory free karo
delete shape1;
delete shape2;

Abstraction - Complex Cheezein Hide Karo! ğŸ™ˆ

Abstraction matlab complex implementation details ko hide karke sirf zaruri features dikhana! Jaise car chalane ke liye engine kaise kaam karta hai janna zaruri nahi!

// Abstract class - Incomplete blueprint
class AbstractEmployee {
protected:
    string name;
    double payRate;

public:
    AbstractEmployee(string n, double p) : name(n), payRate(p) {}

    // Pure virtual function - Ye implement karna compulsory hai!
    virtual double calculatePay() = 0;

    string getName() {
        return name;
    }
};

// Concrete derived classes - Complete blueprints
class HourlyEmployee : public AbstractEmployee {
private:
    int hoursWorked;

public:
    HourlyEmployee(string n, double p, int h) : AbstractEmployee(n, p), hoursWorked(h) {}

    double calculatePay() override {
        return payRate * hoursWorked;  // Hourly calculation
    }
};

class SalariedEmployee : public AbstractEmployee {
public:
    SalariedEmployee(string n, double p) : AbstractEmployee(n, p) {}

    double calculatePay() override {
        return payRate;  // Fixed salary
    }
};

// Using abstraction
HourlyEmployee emp1("John", 15.0, 40);
SalariedEmployee emp2("Jane", 50000.0);

cout << emp1.getName() << " earns $" << emp1.calculatePay() << endl;
cout << emp2.getName() << " earns $" << emp2.calculatePay() << endl;

class Resource {
private:
    int* data;

public:
    // Constructor - Birth time pe resources allocate karo
    Resource() {
        cout << "Resource acquired" << endl;
        data = new int[100];  // Memory allocate kari
    }

    // Destructor - Death time pe resources free karo
    ~Resource() {
        cout << "Resource released" << endl;
        delete[] data;  // Memory free kari
    }
};
