Arrays Modify Karna

<?php
// Indexed array mein elements add karna
$fruits = ["Apple", "Banana"];
$fruits[] = "Cherry";  // End mein "Cherry" add karta hai
print_r($fruits);  // Output: Array ( [0] => Apple [1] => Banana [2] => Cherry )

// Associative array mein elements add karna
$person = [
    "name" => "Rahul",
    "age" => 30
];
$person["city"] = "Mumbai";  // New key-value pair add karta hai
print_r($person);  // Output: Array ( [name] => Rahul [age] => 30 [city] => Mumbai )

// Elements modify karna
$fruits[1] = "Blueberry";  // "Banana" ko "Blueberry" se change karta hai
$person["age"] = 31;       // 30 ko 31 se change karta hai

// Elements remove karna
unset($fruits[0]);         // "Apple" remove karta hai
unset($person["city"]);    // "city" key-value pair remove karta hai

// Unset ke baad array ko re-index karna
$fruits = array_values($fruits);  // Array ko re-index karta hai
print_r($fruits);  // Output: Array ( [0] => Blueberry [1] => Cherry )
?>

Array Operators

<?php
// Union operator (+)
$array1 = ["a" => "Apple", "b" => "Banana"];
$array2 = ["b" => "Blueberry", "c" => "Cherry"];
$result = $array1 + $array2;
print_r($result);  // Output: Array ( [a] => Apple [b] => Banana [c] => Cherry )
// Note: Union operator first array ke keys aur values preserve karta hai

// Equality operator (==)
$array1 = [1, 2, 3];
$array2 = [1, 2, 3];
var_dump($array1 == $array2);  // Output: bool(true)
// Check karta hai ki arrays ke same key-value pairs hai ya nahi

// Identity operator (===)
$array1 = [1, 2, 3];
$array2 = [1, 2, 3];
var_dump($array1 === $array2);  // Output: bool(true)
// Check karta hai ki arrays ke same key-value pairs same order mein aur same types ke hai ya nahi

// Inequality operator (!=)
var_dump($array1 != $array2);  // Output: bool(false)

// Non-identity operator (!==)
var_dump($array1 !== $array2);  // Output: bool(false)
?>

Arrays Mein Loop Lagana

<?php

foreach ($fruits as $fruit) {
    echo $fruit . "<br>";
}
// Output:
// Apple
// Banana
// Cherry

    "name" => "Rahul",
    "age" => 30,
    "city" => "Mumbai"
];

    echo "$key: $value<br>";
}

// name: Rahul
// age: 30
// city: Mumbai

    echo $fruits[$i] . "<br>";
}
// Output:
// Apple
// Banana
// Cherry

// Array_keys ke saath while loop use karna
$person = [
    "name" => "Rahul",
    "age" => 30,
    "city" => "Mumbai"
];
$keys = array_keys($person);
$i = 0;
while ($i < count($keys)) {
    $key = $keys[$i];
    echo "$key: " . $person[$key] . "<br>";
    $i++;
}
// Output:
// name: Rahul
// age: 30
// city: Mumbai
?>

Useful Array Functions ğŸ› ï¸

<?php
// count() - Array mein elements ki count karta hai
$fruits = ["Apple", "Banana", "Cherry"];
echo count($fruits);  // Output: 3

// array_push() - Array ke end mein ek ya more elements add karta hai
array_push($fruits, "Date", "Elderberry");
print_r($fruits);  // Output: Array ( [0] => Apple [1] => Banana [2] => Cherry [3] => Date [4] => Elderberry )

// array_pop() - Array ke last element ko remove aur return karta hai
$lastFruit = array_pop($fruits);
echo $lastFruit;  // Output: Elderberry
print_r($fruits);  // Output: Array ( [0] => Apple [1] => Banana [2] => Cherry [3] => Date )

// array_unshift() - Array ke beginning mein ek ya more elements add karta hai
array_unshift($fruits, "Apricot");
print_r($fruits);  // Output: Array ( [0] => Apricot [1] => Apple [2] => Banana [3] => Cherry [4] => Date )

// array_shift() - Array ke first element ko remove aur return karta hai
$firstFruit = array_shift($fruits);
echo $firstFruit;  // Output: Apricot
print_r($fruits);  // Output: Array ( [0] => Apple [1] => Banana [2] => Cherry [3] => Date )

// array_merge() - Ek ya more arrays ko merge karta hai
$fruits1 = ["Apple", "Banana"];
$fruits2 = ["Cherry", "Date"];
$allFruits = array_merge($fruits1, $fruits2);
print_r($allFruits);  // Output: Array ( [0] => Apple [1] => Banana [2] => Cherry [3] => Date )

// array_slice() - Array ka slice extract karta hai
$fruits = ["Apple", "Banana", "Cherry", "Date", "Elderberry"];
$slice = array_slice($fruits, 1, 3);  // Index 1 se start karke 3 elements get karta hai
print_r($slice);  // Output: Array ( [0] => Banana [1] => Cherry [2] => Date )

// array_splice() - Array ke portion ko remove aur replace karta hai
$fruits = ["Apple", "Banana", "Cherry", "Date", "Elderberry"];
array_splice($fruits, 1, 2, ["Blueberry", "Blackberry"]);  // Index 1 se 2 elements replace karta hai
print_r($fruits);  // Output: Array ( [0] => Apple [1] => Blueberry [2] => Blackberry [3] => Date [4] => Elderberry )

// in_array() - Check karta hai ki value array mein exist karti hai ya nahi
$fruits = ["Apple", "Banana", "Cherry"];
var_dump(in_array("Banana", $fruits));  // Output: bool(true)
var_dump(in_array("Date", $fruits));    // Output: bool(false)

// array_key_exists() - Check karta hai ki key array mein exist karti hai ya nahi
$person = ["name" => "Rahul", "age" => 30];
var_dump(array_key_exists("name", $person));   // Output: bool(true)
var_dump(array_key_exists("email", $person));  // Output: bool(false)

// array_keys() - Array ke saare keys return karta hai
$person = ["name" => "Rahul", "age" => 30, "city" => "Mumbai"];
print_r(array_keys($person));  // Output: Array ( [0] => name [1] => age [2] => city )

// array_values() - Array ke saare values return karta hai
print_r(array_values($person));  // Output: Array ( [0] => Rahul [1] => 30 [2] => Mumbai )

// array_combine() - Ek array ko keys aur dusre ko values use karke array create karta hai
$keys = ["name", "age", "city"];
$values = ["Rahul", 30, "Mumbai"];
$person = array_combine($keys, $values);
print_r($person);  // Output: Array ( [name] => Rahul [age] => 30 [city] => Mumbai )

// array_flip() - Keys aur values ko exchange karta hai
$flipped = array_flip($person);
print_r($flipped);  // Output: Array ( [Rahul] => name [30] => age [Mumbai] => city )

// array_reverse() - Elements ko reverse order mein return karta hai
$fruits = ["Apple", "Banana", "Cherry"];
$reversed = array_reverse($fruits);
print_r($reversed);  // Output: Array ( [0] => Cherry [1] => Banana [2] => Apple )

// sort() - Array ko sort karta hai
$fruits = ["Cherry", "Apple", "Banana"];
sort($fruits);
print_r($fruits);  // Output: Array ( [0] => Apple [1] => Banana [2] => Cherry )

// rsort() - Array ko reverse order mein sort karta hai
$fruits = ["Apple", "Banana", "Cherry"];
rsort($fruits);
print_r($fruits);  // Output: Array ( [0] => Cherry [1] => Banana [2] => Apple )

// asort() - Associative array ko values ke according sort karta hai
$ages = ["Rahul" => 30, "Priya" => 25, "Amit" => 40];
asort($ages);
print_r($ages);  // Output: Array ( [Priya] => 25 [Rahul] => 30 [Amit] => 40 )

// arsort() - Associative array ko values ke according reverse order mein sort karta hai
arsort($ages);
print_r($ages);  // Output: Array ( [Amit] => 40 [Rahul] => 30 [Priya] => 25 )

// ksort() - Associative array ko keys ke according sort karta hai
ksort($ages);
print_r($ages);  // Output: Array ( [Amit] => 40 [Priya] => 25 [Rahul] => 30 )

// krsort() - Associative array ko keys ke according reverse order mein sort karta hai
krsort($ages);
print_r($ages);  // Output: Array ( [Rahul] => 30 [Priya] => 25 [Amit] => 40 )

// array_map() - Array ke har element par callback apply karta hai
$numbers = [1, 2, 3, 4, 5];
$squared = array_map(function($n) {
    return $n * $n;
}, $numbers);
print_r($squared);  // Output: Array ( [0] => 1 [1] => 4 [2] => 9 [3] => 16 [4] => 25 )

// array_filter() - Callback use karke array ke elements filter karta hai
$numbers = [1, 2, 3, 4, 5, 6];
$even = array_filter($numbers, function($n) {
    return $n % 2 == 0;
});
print_r($even);  // Output: Array ( [1] => 2 [3] => 4 [5] => 6 )

// array_reduce() - Callback use karke array ko single value mein reduce karta hai
$numbers = [1, 2, 3, 4, 5];
$sum = array_reduce($numbers, function($carry, $item) {
    return $carry + $item;
}, 0);
echo $sum;  // Output: 15
?>

PHP mein Objects ğŸ§©

Bro, PHP mein objects classes ke instances hote hai, jo objects create karne ke blueprints hote hai. Objects properties (variables) aur methods (functions) contain kar sakte hai.

Objects Create aur Use Karna

<?php
// Class define karo
class Person {
    // Properties
    public $name;
    public $age;
    
    // Constructor
    public function __construct($name, $age) {
        $this->name = $name;
        $this->age = $age;
    }
    
    // Method
    public function greet() {
        return "Hello, mera naam " . $this->name . " hai aur main " . $this->age . " saal ka hoon.";
    }
}

// Object create karo (Person class ka instance)
$person1 = new Person("Rahul", 30);
$person2 = new Person("Priya", 25);

// Properties access karo
echo $person1->name;  // Output: Rahul
echo $person2->age;   // Output: 25

// Methods call karo
echo $person1->greet();  // Output: Hello, mera naam Rahul hai aur main 30 saal ka hoon.
echo $person2->greet();  // Output: Hello, mera naam Priya hai aur main 25 saal ka hoon.
?>

$decodedArray = json_decode($jsonString, true);

?>

Collections Ke Saath Kaam Karne Ke Best Practices ğŸ’¯

  • Right Collection Type Choose Karo: Simple collections ke liye arrays, behavior ke saath complex data ke liye objects, aur specific use cases ke liye SPL data structures use karo.
  • Consistent Keys Use Karo: Associative arrays ke saath kaam karte time, consistent keys use karo taki tumhara code more maintainable bane.
  • Input Validate Karo: Collection mein add karne se pehle hamesha data validate karo, especially jab data user input se aata hai.
  • Type Hints Use Karo: PHP 7+ mein, type hints use karo taki tumhare functions aur methods expected types ke collections receive karein.
  • Collections Document Karo: PHPDoc comments use karke apne arrays aur objects ke structure ko document karo.
  • Performance Consider Karo: Different collection types ke different performance characteristics hote hai. Wo choose karo jo tumhari needs ke liye best fit ho.
  • Built-in Functions Use Karo: PHP arrays ke saath kaam karne ke liye bahut saare built-in functions provide karta hai. Apne khud ke likhne ke bajay unhe use karo.
  • References Ke Saath Careful Raho: Functions ko collections pass karte time, aware raho ki tum value se pass kar rahe ho ya reference se.
